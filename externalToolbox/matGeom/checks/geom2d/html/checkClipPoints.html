% generate 2D points in -50->150 in each direction
points = rand(500, 2)*200 - 50;

% create new figure
figure(1); clf;
axis([-50 150 -50 150]);
hold on;

% draw all points
drawPoint(points, 'b+');

% clip points
box = [0 100 0 100];
clipped = clipPoints(points, box);

% draw clipped points
drawPoint(clipped, 'ko');
drawBox(box);
